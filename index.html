import re # Read the file with open('c:/Users/Admin/Downloads/sign-main (2)/sign-main/index.html', 'r', encoding='utf-8') as f: content = f.read() # 1. Add nav item nav_item = """
  • Deepti Mehta (Sustainability Manager)
  • Preeti Rani
    • """ content = content.replace( """
  • Deepti Mehta (Sustainability Manager)
    • """, nav_item ) # 2. Extract Deepti's template tpl_match = re.search(r"()", content, re.DOTALL) if not tpl_match: print("Could not find Deepti's template") exit(1) deepti_tpl = tpl_match.group(1) # 3. Create Preeti's template by replacing values preeti_tpl = deepti_tpl.replace("tpl-deepti", "tpl-preeti") preeti_tpl = preeti_tpl.replace("Deepti Mehta Email Signature", "Preeti Rani Email Signature") preeti_tpl = preeti_tpl.replace( "https://github.com/learnDataScienceIndia/signature/blob/71419008b56cd5d90cc38924a7b25330b738b8d6/Screenshot%202026-06-16%20151709.png?raw=true", "https://raw.githubusercontent.com/learnDataScienceIndia/signature/54e0197b89fcf5b9e21c3824f634082238383f13/mam%20photo.png" ) preeti_tpl = preeti_tpl.replace("alt=\"Deepti Mehta\"", "alt=\"Preeti Rani\"") preeti_tpl = preeti_tpl.replace( "Deepti Mehta", "Preeti Rani" ) preeti_tpl = preeti_tpl.replace( "Head of Environmental Compliance & Sustainability", "Project Manager" ) preeti_tpl = preeti_tpl.replace( "href=\"tel:+971544103099\"", "href=\"tel:0543252882\"" ) preeti_tpl = preeti_tpl.replace( "+971 54 410 3099", "0543252882" ) preeti_tpl = preeti_tpl.replace( "href=\"mailto:deepti@reloopglobal.com\"", "href=\"mailto:preeti@reloopglobal.com\"" ) preeti_tpl = preeti_tpl.replace( "deepti@reloopglobal.com", "preeti@reloopglobal.com" ) preeti_tpl = preeti_tpl.replace( "href=\"https://wa.me/971544103099\"", "href=\"https://wa.me/971543252882\"" # Assumed country code +971 since UAE ) # 4. Insert Preeti's template right after Deepti's template content = content.replace(deepti_tpl, deepti_tpl + "\n\n" + preeti_tpl) # Save the modified content back with open('c:/Users/Admin/Downloads/sign-main (2)/sign-main/index.html', 'w', encoding='utf-8') as f: f.write(content) print("Successfully updated index.html")
